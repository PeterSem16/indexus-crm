 same => n,GotoIf($["${EXTEN:0:2}"="36"]?route-hu,${EXTEN},1)
 same => n,GotoIf($["${EXTEN:0:2}"="40"]?route-ro,${EXTEN},1)
 same => n,GotoIf($["${EXTEN:0:3}"="421"]?route-sk,${EXTEN},1)
